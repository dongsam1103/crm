function doGet(e){return h(e)}function doPost(e){return h(e)}
function h(e){var lk=LockService.getScriptLock();lk.tryLock(10000);
try{var p=e.parameter,r;
if(p.action=='read')r=rd(p.sheet);
else if(p.action=='write')r=wr(p.sheet,JSON.parse(p.data));
else if(p.action=='ping')r={ok:true,t:new Date().toISOString()};
else r={error:'unknown'};
return ContentService.createTextOutput(JSON.stringify(r)).setMimeType(ContentService.MimeType.JSON);
}catch(e){return ContentService.createTextOutput(JSON.stringify({error:e.message})).setMimeType(ContentService.MimeType.JSON);}
finally{lk.releaseLock();}}
function rd(n){var ss=SpreadsheetApp.getActiveSpreadsheet(),sh=ss.getSheetByName(n);
if(!sh){ss.insertSheet(n);return{data:[]};}
var v=sh.getDataRange().getValues();if(v.length<2)return{data:[]};
var hd=v[0];return{data:v.slice(1).map(function(r){var o={};hd.forEach(function(k,i){o[k]=r[i]||''});return o})};}
function wr(n,data){var ss=SpreadsheetApp.getActiveSpreadsheet(),sh=ss.getSheetByName(n);
if(!sh)sh=ss.insertSheet(n);sh.clearContents();
if(!data||!data.length)return{ok:true};
var h=Object.keys(data[0]);sh.getRange(1,1,1,h.length).setValues([h]);
var rows=data.map(function(obj){return h.map(function(k){return obj[k]||''})});
if(rows.length)sh.getRange(2,1,rows.length,h.length).setValues(rows);
return{ok:true,rows:rows.length};}
